# Weight-Rating function
def weight_rating(x , m=M,C=VoteAvg):
    v = x['vote_count']
    R = x['vote_average']
    
    rating = (v/(v+m) * R) + (m/(m+v) * C)
    
    return rating

def get_recommendations(title, cos_sim=cos_sim):
   
    title = indices[title]
   
    sim_scores_list = list(enumerate(cos_sim[title]))
    
    sim_scores_list = sorted(sim_scores_list, key=lambda x: x[1], reverse=True)
  
    sim_scores_list = sim_scores_list[1:11]
  
    movie_recommended = [i[0] for i in sim_scores_list]

    return Data['title'].iloc[movie_recommended]
